20       @BeforeClass
21       @Parameters("Browser")

23           if (BrowserName.equalsIgnoreCase("Chrome")) {
24               AlyDriver = new ChromeDriver();
25   
26           } else if (BrowserName.equalsIgnoreCase("Firefox")) {
27               AlyDriver = new FirefoxDriver();
28           } else if (BrowserName.equalsIgnoreCase("IE")) {
29               AlyDriver = new InternetExplorerDriver();
30           } else if (BrowserName.equalsIgnoreCase("Safari")) {
31               AlyDriver = new SafariDriver();
32           } else if (BrowserName.equalsIgnoreCase("Edge")) {
33               AlyDriver = new EdgeDriver();
34           }
35   
36           
37           AlyDriver.manage().timeouts().implicitlyWait(Duration.ofSeconds(5));
38   
39           AlyDriver.navigate().to("https://opensource-demo.orangehrmlive.com/");
40   
41           AlyDriver.manage().window().maximize();
42   
43       }
